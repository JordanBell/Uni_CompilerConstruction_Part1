foo(x) {x+1}

main(count)
{
	// Test while loop
	new total = 0 in
	new input = 0 in

	// Based on the input argument, add the total of all input ints
	while (~count >= 1)
	{
		input = read_int();
		total = ~total + foo(input);
		count = ~count - 1
	};

	// Print and return the total
	print_int (5);
	~total
}
